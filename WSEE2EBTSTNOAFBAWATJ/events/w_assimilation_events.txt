country_event = {
	id = w_assimilation_events.13
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		every_owned_pop = {
			if = {
				limit = {
					OR = {
						has_job = institutionalized_sadistics
						has_job = institutionalized_sanes
						has_job = institutionalized_spiritualists
						has_job = institutionalized_materialists
						has_job = institutionalized_egalitarians
						has_job = institutionalized_authoritarians
						has_job = institutionalized_xenophobes
						has_job = institutionalized_xenophiles
						has_job = institutionalized_militarists
						has_job = institutionalized_pacifists
					}
				}
				clear_ethos = yes
				pop_change_ethic = ethic_sadistic
				unemploy_pop = yes
			}
		}
		every_owned_planet = {
			if = {
				limit = {
					has_modifier = "w_add_institutionalized_sadistics"
				}
				remove_modifier = "w_add_institutionalized_sadistics"
			}
