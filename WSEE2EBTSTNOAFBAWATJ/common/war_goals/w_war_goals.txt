wg_w_harvest = {
	casus_belli = cb_w_harvest
	defender_default = yes 
	threat_multiplier = 1.0
	hide_if_no_cb = yes
	cede_claims = no
	war_exhaustion = 1.0 
	show_claims_in_description = no
	release_occupied_systems_on_status_quo = no
	potential = { always = yes }
	possible = { always = yes }
	ai_weight = { weight = 1 }
	allowed_peace_offers = { status_quo surrender demand_surrender }
	surrender_acceptance = -40
	total_war = no
	war_exhaustion = 1.5

	on_status_quo = {
		custom_tooltip = war_goal_wg_w_harvest_tooltip
		root = {
			every_controlled_planet = {
				limit = { 
					is_colony = yes
					is_owned_by = from
				} 
				add_modifier = { modifier = "w_harvested" days = 1800 } # 5 years of pop malus
				# Get 200 food for every pop on this planet
				hidden_effect = { 
					every_owned_pop = { 
						limit = { is_robotic = no }
						root = { add_resource = { food = 300 } } 
					}
				}
			}
		}
	}

	on_accept = {
		custom_tooltip = war_goal_wg_w_harvest_tooltip
		from = {
			every_owned_planet = {
				add_modifier = { modifier = "w_harvested" days = 1800 } # 5 years of pop malus
				# Get 200 food for every pop on this planet
				hidden_effect = { 
					every_owned_pop = {
						limit = { is_robotic = no }
						root = { add_resource = { food = 300 } } 
					}
				}
			}
		}
	}
}
